import random
import math

def gerar_equacao():
    # Coeficientes aleatórios entre -10 e 10
    # O 'a' não pode ser zero, senão vira uma equação do primeiro grau
    a = random.choice([i for i in range(-10, 11) if i != 0])
    b = random.randint(-10, 10)
    c = random.randint(-10, 10)

    # Formatação visual (ajustando os sinais)
    sinal_b = "+" if b >= 0 else ""
    sinal_c = "+" if c >= 0 else ""
    
    print(f"Sua equação é: {a}x² {sinal_b}{b}x {sinal_c}{c} = 0")

    # Cálculo do Delta
    delta = b**2 - 4*a*c
    print(f"Delta (Δ) = {delta}")

    if delta < 0:
        print("A equação não possui raízes reais.")
    else:
        x1 = (-b + math.sqrt(delta)) / (2*a)
        x2 = (-b - math.sqrt(delta)) / (2*a)
        print(f"Raízes: x1 = {x1:.2f}, x2 = {x2:.2f}")

# Executar a função
gerar_equacao()
